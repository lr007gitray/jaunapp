function contemLink(texto){

  texto = texto.toLowerCase();

  return texto.includes("http")
      || texto.includes("www")
      || texto.includes(".com")
      || texto.includes("<script")
      || texto.includes("</script")
      || texto.includes("javascript:");
}

function enviarInterno(){

  let campos = document.querySelectorAll("#formInterno input, #formInterno select");

  for(let campo of campos){

    if(contemLink(campo.value)){

      alert("Links ou scripts não são permitidos.");

      return;
    }
  }

  let texto = "📋 Pesquisa Interna IASD\n\n";

  campos.forEach(c => {
    texto += (c.placeholder || c.previousElementSibling.innerText) + ": " + c.value + "\n";
  });

  window.open(
    "https://wa.me/5511943408434?text=" + encodeURIComponent(texto)
  );
}
